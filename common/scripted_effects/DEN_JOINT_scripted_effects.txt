DEN_JOINT_give_cores_on_chinese_states = {
    if = {
        limit = {
            any_state = {
                DEN_is_chinese_state = yes
                NOT = { is_core_of = ROOT }
            }
        }
        custom_effect_tooltip = DEN_JOINT_give_cores_on_chinese_states_tt
    }
    hidden_effect = {
        every_state = {
			limit = {
				DEN_is_chinese_state = yes
                NOT = { is_core_of = ROOT }
			}
			add_core_of = ROOT
		}
    }
    log = "ROOT: [ROOT.GetName] gained cores on all chinese states"
}
